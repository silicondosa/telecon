68class MySIN : public mpFX
69{
70 double m_freq, m_amp;
71public:
72 MySIN(double freq, double amp) : mpFX(wxT("f(x) = SIN(x)"), mpALIGN_LEFT) { m_freq = freq; m_amp = amp; m_drawOutsideMargins = false; }
73 virtual double GetY(double x) { return m_amp * sin(x / 6.283185 / m_freq); }
74 virtual double GetMinY() { return -m_amp; }

82 double m_freq, m_amp;
83public:
84 MyCOSinverse(double freq, double amp) : mpFY(wxT("g(y) = COS(y)"), mpALIGN_BOTTOM) { m_freq = freq; m_amp = amp; m_drawOutsideMargins = false; }
85 virtual double GetX(double y) { return m_amp * cos(y / 6.283185 / m_freq); }

92class MyLissajoux : public mpFXY
93{
94 double m_rad;
95 int m_idx;
96public:
97 MyLissajoux(double rad) : mpFXY(wxT("Lissajoux")) { m_rad = rad; m_idx = 0; m_drawOutsideMargins = false; }
98 virtual bool GetNextXY(double& x, double& y)
99 {
100 if (m_idx < 360)
101 {
102 x = m_rad * cos(m_idx / 6.283185 * 360);
103 y = m_rad * sin(m_idx / 6.283185 * 360 * 3);
104 m_idx++;
105 return TRUE;
106 }
107 else
108 {
109 return FALSE;
110 }
111 }
112 virtual void Rewind() { m_idx = 0; }
113 virtual double GetMinX() { return -m_rad; }
114 virtual double GetMaxX() { return m_rad; }
115 virtual double GetMinY() { return -m_rad; }
116 virtual double GetMaxY() { return m_rad; }
117};
