226void TeleconRealTimeLineChart::DrawChart()
227{
228 // Create an XYChart object 600 x 270 pixels in size, with light grey (f4f4f4)
229 // background, black (000000) border, 1 pixel raised effect, and with a rounded frame.
230 XYChart *c = new XYChart(520, 270, 0xf4f4f4, 0x000000, 1);
231 c->setRoundedFrame((m_bgColour.Red() << 16) + (m_bgColour.Green() << 8) + m_bgColour.Blue());
232
233 // Set the plotarea at (55, 55) and of size 520 x 185 pixels. Use white (ffffff)
234 // background. Enable both horizontal and vertical grids by setting their colors to
235 // grey (cccccc). Set clipping mode to clip the data lines to the plot area.
236 c->setPlotArea(55, 55, 520, 185, 0xffffff, -1, -1, 0xcccccc, 0xcccccc);
237 c->setClipping();
238
239 // Add a title to the chart using 15 pts Times New Roman Bold Italic font, with a light
240 // grey (dddddd) background, black (000000) border, and a glass like raised effect.
241 c->addTitle(m_chartTitle, "Times New Roman Bold Italic", 15)->setBackground(0xdddddd, 0x000000, Chart::glassEffect());
242
243 // Set the reference font size of the legend box
244 c->getLegend()->setFontSize(8);
245
246 // Configure the y-axis with a 10pts Arial Bold axis title
247 c->yAxis()->setTitle(m_ylabel, "Arial Bold", 10);
248
249 // Configure the x-axis to auto-scale with at least 75 pixels between major tick and
250 // 15 pixels between minor ticks. This shows more minor grid lines on the chart.
251 c->xAxis()->setTickDensity(75, 15);
252
253 // Set the axes width to 2 pixels
254 c->xAxis()->setWidth(2);
255 c->yAxis()->setWidth(2);
256
257 // Now we add the data to the chart.
258 if (m_timeStamps.size() > 0)
259 {
260 double firstTime = m_timeStamps[0];
261 // Set up the x-axis to show the time range in the data buffer
262 c->xAxis()->setDateScale(firstTime, firstTime + dataInterval * sampleSize / 1000);
263
264 // Set the x-axis label format
265 c->xAxis()->setLabelFormat("{value|hh:nn:ss}");
266
267 // Create a line layer to plot the lines
268 LineLayer *layer = c->addLineLayer();
269
270 // The x-coordinates are the timeStamps.
271 layer->setXData(DoubleArray(&m_timeStamps[0], m_timeStamps.size()));
272
273 // The data series are used to draw lines.
274 for (const auto &plot : m_plots)
275 {
276 if (plot->getLineType() == LT_SOLID)
277 {
278 layer->addDataSet(DoubleArray(&(plot->getOldest()), plot->size()), plot->getColor(), plot->getPlotTitle().c_str());
279 }
280 else if (plot->getLineType() == LT_SCATTER)
281 {
282 c->addScatterLayer(DoubleArray(&m_timeStamps[0], m_timeStamps.size()),
283 DoubleArray(&(plot->getOldest()), plot->size()), plot->getPlotTitle().c_str(), 1, 5, plot->getColor());
284 }
285 // c->addScatterLayer(DoubleArray(&m_timeStamps[0], m_timeStamps.size()), DoubleArray(&(plot->getOldest()), plot->size()), plot->getPlotTitle().c_str(), 1, 5, plot->getColor());
286 }
287 }
288
289 // Include track line with legend. If the mouse is on the plot area, show the track
290 // line with legend at the mouse position; otherwise, show them for the latest data
291 // values (that is, at the rightmost position).
292 TrackLineLegend(c, m_chartViewer->isMouseOnPlotArea() ? m_chartViewer->getPlotAreaMouseX() : c->getPlotArea()->getRightX());
293
294 if (m_chartViewer->getChart() != NULL)
295 {
296 delete m_chartViewer->getChart();
297 }
298 // Set the chart image to the WinChartViewer
299 m_chartViewer->setChart(c);
300}

309// Draw the track line with legend
310void TeleconRealTimeLineChart::TrackLineLegend(XYChart *c, int mouseX)
311{
312 // Clear the current dynamic layer and get the DrawArea object to draw on it.
313 DrawArea *d = c->initDynamicLayer();
314
315 // The plot area object
316 PlotArea *plotArea = c->getPlotArea();
317
318 // Get the data x-value that is nearest to the mouse, and find its pixel coordinate.
319 double xValue = c->getNearestXValue(mouseX);
320 int xCoor = c->getXCoor(xValue);
321
322 // Draw a vertical track line at the x-position
323 d->vline(plotArea->getTopY(), plotArea->getBottomY(), xCoor, d->dashLineColor(0x000000, 0x0101));
324
325 // Container to hold the legend entries
326 vector<string> legendEntries;
327
328 // Iterate through all layers to build the legend array
329 for (int i = 0; i < c->getLayerCount(); ++i)
330 {
331 Layer *layer = c->getLayerByZ(i);
332
333 // The data array index of the x-value
334 int xIndex = layer->getXIndexOf(xValue);
335
336 // Iterate through all the data sets in the layer
337 for (int j = 0; j < layer->getDataSetCount(); ++j)
338 {
339 DataSet *dataSet = layer->getDataSetByZ(j);
340
341 // We are only interested in visible data sets with names
342 const char *dataName = dataSet->getDataName();
343 int color = dataSet->getDataColor();
344 if (dataName && *dataName && (color != (int)Chart::Transparent))
345 {
346 // Build the legend entry, consist of the legend icon, name and data value.
347 double dataValue = dataSet->getValue(xIndex);
348 ostringstream legendEntry;
349 legendEntry << "<*block*>" << dataSet->getLegendIcon() << " " << dataName << ": " << ((dataValue == Chart::NoValue) ? "N/A" : c->formatValue(dataValue, "{value|P4}"))
350 << "<*/*>";
351 legendEntries.push_back(legendEntry.str());
352
353 // Draw a track dot for data points within the plot area
354 int yCoor = c->getYCoor(dataSet->getPosition(xIndex), dataSet->getUseYAxis());
355 if ((yCoor >= plotArea->getTopY()) && (yCoor <= plotArea->getBottomY()))
356 {
357 d->circle(xCoor, yCoor, 4, 4, color, color);
358 }
359 }
360 }
361 }
362
363 // Create the legend by joining the legend entries
364 ostringstream legendText;
365 legendText << "<*block,maxWidth=" << plotArea->getWidth() << "*><*block*><*font=Arial Bold*>["
366 << c->xAxis()->getFormattedLabel(xValue, "hh:nn:ss") << "]<*/*>";
367 for (int i = ((int)legendEntries.size()) - 1; i >= 0; --i)
368 {
369 legendText << " " << legendEntries[i];
370 }
371
372 // Display the legend on the top of the plot area
373 TTFText *t = d->text(legendText.str().c_str(), "Arial", 8);
374 t->draw(plotArea->getLeftX() + 5, plotArea->getTopY() - 3, 0x000000, Chart::BottomLeft);
375 t->destroy();
376}
